//Vscript

// 리트랙션 파라미터
double dz = 1.0;   // mm 단위, 위로 들 높이

// CN16 P0 신호 읽기 (0 = 눌림, 1 = 평상시)
int sw = WINC_INPUT(0);   // P0은 index 0

if (sw == 0) {
    // 스위치 눌림 → 리트랙션 실행
    WINC_CMD("mdi", "G91");                            // 상대모드
    WINC_CMD("mdi", format("G0 Z%0.3f", dz));          // 위로 dz
    WINC_CMD("mdi", "G90");                            // 절대모드 복귀
}
